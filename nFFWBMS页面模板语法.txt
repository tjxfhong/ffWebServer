<!--{ }-->		语句起始符
+ - * / %		算术运算符
> >= < <= == !=	比较运算符
&& || ！		逻辑运算符
ffwbms_all	查看所有变量
ffwbms_see $变量名	查看指定变量的值
$变量名		页面中使用变量的值
$变量名=值	给变量赋值
include "文件名"	引入文件，文件如果带地址，应该加上相对地址
ffwbms 扩展名,返回变量名,"参数"	参数为名=值对，多个时用|符分隔，可以用连接符连接字符串和变量
				参数中有md5时，里面的变量不写$，例：
				<!--{ffwbms ffa_get_record_test,vtemp,'id='.$get["id"]."|s1=".md5(post["c1"])}-->
if 值 比较 值	值可以是变量，也可以是字符，if无法处理内部带语法的语句，后期重写语法树解析
	if  elseif  else  /if
loop $变量 $临时变量
	loop /loop

setcookies 键，值	键加不加引号都可，值同理
	<!--{setcookies "c1",$post["s1"]}-->
	<!--{setcookies "c2","s测试网页2"}-->
getcookie 键	键加不加引号都可
	c1: <!--{if getcookie[uid]==""}--> 
	c2: <!--{getcookie(uid)}--><br /><br />

赋值支持+，-，*，/，.，的二元运算，如：
	<!--{$id1["a"]=1011}--><!--{$id2["b"]=1011}--><br>
	<!--{if $id1["a"]==$id2["b"]}-->(<!--{$id1["a"]}-->|<!--{$id2["b"]}-->)相等<!--{else}-->(<!--{$id1["a"]}-->|	<!--{$id2["b"]}-->)不相等<!--{/if}-->|<br>
	<!--{$id1a="kkk=".$id2["b"]}--><!--{$id1a}-->|<br>
	<!--{$ia=1}--><!--{$ia=$ia+1}--><!--{$ia=$ia+1}--><!--{$ia=$ia+1}--><!--{$ia=$ia+1}--><br>
	(<!--{$ia}-->)<br>
	<!--{$ib=$ia-1}--><!--{$ib=$ib-1}-->
	(<!--{$ib}-->)<br>
	<!--{$ic=$ia*$ib}-->
	(<!--{$ic}-->)<br>
	<!--{$id=$ic/14}-->
	(<!--{$id}-->)<br>
